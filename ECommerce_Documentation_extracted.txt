1. Project Overview
This documentation covers the end-to-end development of a full-stack e-commerce web application. The platform enables customers to browse products, manage carts, place orders, and process payments — while providing administrators with tools to manage inventory, orders, and users.
1.1 Goals & Objectives
Build a responsive, performant storefront accessible on all devices
Implement secure authentication with role-based access control
Provide a robust REST API using ASP.NET Core
Store all data reliably in a PostgreSQL relational database
Support product catalog, cart, checkout, and order management
Enable admin dashboard for inventory and order tracking

3. System Architecture
3.1 High-Level Architecture
The application follows a clean client-server architecture with a clear separation between the React SPA frontend and the ASP.NET Core Web API backend.
Pattern
Layered Architecture: Presentation → API → Business Logic → Data Access → Database
3.2 Architecture Layers
Frontend (React SPA)
Pages / Views — Route-level components (ProductListPage, CartPage, CheckoutPage)
Components — Reusable UI pieces (ProductCard, Navbar, Modal, Button)
Hooks — Custom React hooks for shared logic (useCart, useAuth, useProducts)
Store (Zustand) — Global client state (cart items, auth user, UI state)
Services — Axios API call wrappers (productService, authService, orderService)
Backend (ASP.NET Core Web API)
Controllers — HTTP request handlers, route definitions
Services — Business logic and orchestration
Repositories — Data access patterns abstracting EF Core queries
Models / Entities — EF Core entity classes mapped to DB tables
DTOs — Data Transfer Objects for API request/response shaping
Middleware — Auth, logging, exception handling, CORS
3.3 Data Flow
User interacts with React UI (clicks 'Add to Cart')
React component dispatches action or calls service function
Axios sends HTTP request to ASP.NET Core API with JWT in Authorization header
API middleware validates JWT, passes to controller
Controller delegates to Service layer for business logic
Service calls Repository, which queries PostgreSQL via EF Core
Data flows back as DTO → JSON → React state → re-render

headers: { 'Content-Type': 'application/json' }

return config
})
// Handle 401 — redirect to login

res => res,
err => {

}

[ApiController]
[Route("api/[controller]")]
public class ProductsController : ControllerBase
{
private readonly IProductService _productService;
public ProductsController(IProductService productService)
=> _productService = productService;
[HttpGet]
public async Task<IActionResult> GetAll(
[FromQuery] string? search,
[FromQuery] Guid? categoryId,
[FromQuery] decimal? minPrice,
[FromQuery] decimal? maxPrice,
[FromQuery] string? sortBy = "createdAt",
[FromQuery] int page = 1,
[FromQuery] int pageSize = 12)
{

search, categoryId, minPrice, maxPrice, sortBy, page, pageSize);
return Ok(result);
}
[HttpGet("{slug}")]
public async Task<IActionResult> GetBySlug(string slug)
{
var product = await _productService.GetBySlugAsync(slug);
return product == null ? NotFound() : Ok(product);
}
[HttpPost]
[Authorize(Roles = "Admin")]
public async Task<IActionResult> Create([FromBody] CreateProductDto dto)
{
var product = await _productService.CreateAsync(dto);
return CreatedAtAction(nameof(GetBySlug), new { slug = product.Slug }, product);
}
[HttpPut("{id}")]
[Authorize(Roles = "Admin")]
public async Task<IActionResult> Update(Guid id, [FromBody] UpdateProductDto dto)
{
await _productService.UpdateAsync(id, dto);
return NoContent();
}
[HttpDelete("{id}")]
[Authorize(Roles = "Admin")]
public async Task<IActionResult> Delete(Guid id)
{
await _productService.DeleteAsync(id);
return NoContent();
}
}

8.2 Shopping Cart
Zustand Cart Store
// src/store/cartStore.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
interface CartItem { id: string; name: string; price: number; qty: number; imageUrl: string }
interface CartStore {
items: CartItem[]
addItem: (item: Omit<CartItem, 'qty'>) => void
removeItem: (id: string) => void
updateQty: (id: string, qty: number) => void
clearCart: () => void
total: () => number
}
export const useCartStore = create<CartStore>()(persist(
(set, get) => ({
items: [],
addItem: (item) => set(state => {
const existing = state.items.find(i => i.id === item.id)
if (existing) return { items: state.items.map(i => i.id === item.id ? { ...i, qty: i.qty + 1 } : i) }
return { items: [...state.items, { ...item, qty: 1 }] }
}),
removeItem: (id) => set(state => ({ items: state.items.filter(i => i.id !== id) })),
updateQty: (id, qty) => set(state => ({ items: state.items.map(i => i.id === id ? { ...i, qty } : i) })),
clearCart: () => set({ items: [] }),
total: () => get().items.reduce((sum, i) => sum + i.price * i.qty, 0)
}),
{ name: 'cart-storage' }
))

9. Authentication & Authorization
9.1 Auth Service (Backend)
// Application/Services/AuthService.cs
public class AuthService : IAuthService
{
private readonly UserManager<User> _userManager;
private readonly IConfiguration _config;
public async Task<AuthResponseDto> LoginAsync(LoginDto dto)
{
var user = await _userManager.FindByEmailAsync(dto.Email)
?? throw new UnauthorizedException("Invalid credentials");
if (!await _userManager.CheckPasswordAsync(user, dto.Password))
throw new UnauthorizedException("Invalid credentials");
var token = GenerateJwtToken(user);
return new AuthResponseDto { Token = token, User = MapToDto(user) };
}
private string GenerateJwtToken(User user)
{
var claims = new[]
{
new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
new Claim(ClaimTypes.Email, user.Email!),
new Claim(ClaimTypes.Role, "Customer") // or Admin
};
var key  = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["Jwt:Key"]!));
var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
var token = new JwtSecurityToken(
issuer:   _config["Jwt:Issuer"],
audience: _config["Jwt:Audience"],
claims:   claims,
expires:  DateTime.UtcNow.AddDays(7),
signingCredentials: creds);
return new JwtSecurityTokenHandler().WriteToken(token);
}
}
9.2 Auth Store (Frontend)
// src/store/authStore.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
interface AuthStore {
token: string | null
user: { id: string; email: string; role: string } | null
login: (token: string, user: AuthStore['user']) => void
logout: () => void
isAdmin: () => boolean
}
export const useAuthStore = create<AuthStore>()(persist(
(set, get) => ({
token: null,
user: null,
login: (token, user) => set({ token, user }),
logout: () => set({ token: null, user: null }),
isAdmin: () => get().user?.role === 'Admin'
}),
{ name: 'auth-storage' }
))
9.3 Protected Route Component
// src/components/ProtectedRoute.tsx
import { Navigate, Outlet } from 'react-router-dom'
import { useAuthStore } from '../store/authStore'
interface Props { adminOnly?: boolean }
export default function ProtectedRoute({ adminOnly = false }: Props) {
const { token, isAdmin } = useAuthStore()
if (!token) return <Navigate to='/login' replace />
if (adminOnly && !isAdmin()) return <Navigate to='/' replace />
return <Outlet />
}

// API/Controllers/PaymentsController.cs
[ApiController]
[Route("api/[controller]")]
[Authorize]
public class PaymentsController : ControllerBase
{
[HttpPost("create-intent")]
public async Task<IActionResult> CreatePaymentIntent([FromBody] PaymentIntentDto dto)
{
StripeConfiguration.ApiKey = _config["Stripe:SecretKey"];
var options = new PaymentIntentCreateOptions
{
Amount   = (long)(dto.Amount * 100), // Convert to cents
Currency = "usd",
AutomaticPaymentMethods = new PaymentIntentAutomaticPaymentMethodsOptions { Enabled = true }
};
var service = new PaymentIntentService();
var intent  = await service.CreateAsync(options);
return Ok(new { clientSecret = intent.ClientSecret });
}
[HttpPost("webhook")]
[AllowAnonymous]
public async Task<IActionResult> Webhook()
{
var payload   = await new StreamReader(Request.Body).ReadToEndAsync();
var signature = Request.Headers["Stripe-Signature"];
var evt = EventUtility.ConstructEvent(payload, signature, _config["Stripe:WebhookSecret"]);
if (evt.Type == Events.PaymentIntentSucceeded)
{
// Mark order as paid, trigger fulfillment
}
return Ok();
}
}

// src/components/checkout/PaymentForm.tsx
import { PaymentElement, useStripe, useElements } from '@stripe/react-stripe-js'
export default function PaymentForm({ onSuccess }: { onSuccess: () => void }) {
const stripe   = useStripe()
const elements = useElements()
const handleSubmit = async (e: React.FormEvent) => {
e.preventDefault()
if (!stripe || !elements) return
const { error } = await stripe.confirmPayment({
elements,
confirmParams: { return_url: window.location.origin + '/order-success' }
})
if (error) console.error(error.message)
else onSuccess()
}
return (
<form onSubmit={handleSubmit} className='space-y-4'>
<PaymentElement />
<button type='submit' className='w-full bg-primary-600 text-white py-3 rounded-lg font-semibold hover:bg-primary-700 transition'>
Pay Now
</button>
</form>
)
}

Install Nginx and configure reverse proxy for /api and / routes
